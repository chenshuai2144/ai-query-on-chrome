如何将Less代码转换为CSS in JS？

```typescript
import { useEmotionCss } from "@emotion/css";

const AppLogo = () => {
  return (
    <div className={useEmotionCss(() => ({
      height: "40vmin",
      pointerEvents: "none",
      "@keyframes spin": {
        from: {
          transform: "rotate(0deg)",
        },
        to: {
          transform: "rotate(360deg)",
        },
      },
      animationName: "spin",
      animationDuration: "20s",
      animationTimingFunction: "linear",
      animationIterationCount: "infinite",
    }))}
  );
};

const AppHeader = () => {
  return (
    <div className={useEmotionCss(() => ({
      backgroundColor: "#282c34",
      minHeight: "100vh",
      display: "flex",
      flexDirection: "column",
      alignItems: "center",
      justifyContent: "center",
      fontSize: "calc(10px + 2vmin)",
      color: "white",
    }))}
  );
};

const AppLink = () => {
  return (
    <div className={useEmotionCss(() => ({
      color: "#61dafb",
    }))}
  );
};

export { AppLogo, AppHeader, AppLink };
```
