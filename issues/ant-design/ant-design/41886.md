import "./index.css";
import { Select } from "antd";

const option = [];
for (let i = 0; i < 1000; i  ) {
  option.push({
    value: `jack${i}`,
    label: (
      <div
        style={{
          width: 100,
          height: 50,
          display: "flex",
          justifyContent: "center",
          alignItems: "center",
          border: "1px solid red"
        }}
        >
        jack{i}
      </div>
    )
  });
}
const App: React.FC = () => (
  <Select
    style={{ width: 500 }}
    options={option}
    // listItemHeight 不管写多少永远只有6个 写0是全部
    listItemHeight={0}
    dropdownRender={(dom) => {
      return <div className="dropdown-render">{dom}</div>;
    }}
    />
);

export default App;



.dropdown-render .rc-virtual-list-holder-inner {

}
