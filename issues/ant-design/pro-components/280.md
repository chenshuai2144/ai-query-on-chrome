const serialColumnMake = ({
  current = 1,
  pageSize = 10,
  width = 50,
  fixed = false,
} = {}) => {
  if (current && pageSize) {

      // eslint-disable-next-line react/react-in-jsx-scope

    };

    dataIndex: "serial",
    width,
    fixed,
    className: "ant-table-serial-column",
    render: (text, row, index) => <span>{index + 1}</span>,
  };
};
