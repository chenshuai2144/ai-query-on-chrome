### 💻 复现代码

const columns: ProColumns<any>[] = [
{
title: '验收结果',
dataIndex: 'notify',
valueType: 'radio',
valueEnum: {
1: {
text: '合格',
},
2: {
text: '不合格',
},
},
formItemProps: {
rules: [
{
required: true,
message: '此项为必填项',
},
],
},
},
]
// 表格部分
<EditableProTable<any>
columns={columns}
className="editableProTableIndex"
rowKey="id"
value={secondInfo}
onChange={setSecondInfo}
pagination={false}
recordCreatorProps={{
              style: {
                display: 'none',
              },
              record: () => ({
                key: Date.now(),
              }),
            }}
editable={{
              type: 'multiple',
              editableKeys,
              form: editForm,
              // actionRender: (row, config, defaultDoms) => {
              //   return [defaultDoms.delete]
              // },
              onValuesChange: (record, recordList) => {
                setSecondInfo(recordList)
              },
              onChange: setEditableRowKeys,
            }}
/>
