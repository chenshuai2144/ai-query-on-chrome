### 💻 复现代码

`<ModalForm<OrderFormInfo> grid={true}>
        <ProForm.Group>
                <ProFormDateTimePicker
                        style={{
                            width: '332px'
                        }}
                        label="下单时间"
                        colProps={{xl: 12}}
                        name="orderTime"
                        required
                        grid={true}
                    }}
                />
                <ProFormText disabled colProps={{xl: 12}} name="pay" label="支付方式"/>
        </ProForm.Group>
</ModalForm>`
