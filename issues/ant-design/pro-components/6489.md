这个问题可能是由于filter组件没有正确地将表单值传递给request函数。你可以在filter组件中添加一个额外的属性，例如`formRef`，然后在请求函数中使用这个属性来获取表单值。例如：

```jsx
import React, { useState } from "react";
import { Form, Input, Button } from "antd";
import { ProTable, ProForm } from "@ant-design/pro-components";

const columns = [
  {
    title: "Name",
    dataIndex: "name",
    filters: [
      {
        text: "Filter 1",
        value: "filter1",
      },
      {
        text: "Filter 2",
        value: "filter2",
      },
    ],
  },
];

const App = () => {
  const [form] = Form.useForm();
  const [data, setData] = useState([]);

  const handleSearch = (value) => {
    setData(data.filter((item) => item.name.includes(value)));
  };

  const handleReset = () => {
    setData([]);
  };

  const handleAdd = () => {
    const newItem = {
      name: `New ${data.length + 1}`,
    };
    setData([...data, newItem]);
  };

  const handleSubmit = (values) => {
    console.log("Received values of form: ", values);
  };

  return (
    <Form form={form}>
      <Form.Item name="search">
        <Input placeholder="Search" />
      </Form.Item>
      <Form.Item name="filter">
        <ProForm.Select
          placeholder="Filter"
          style={{ width: 200 }}
          form={form}
          options={[
            {
              label: "Option 1",
              value: "option1",
            },
            {
              label: "Option 2",
              value: "option2",
            },
          ]}
        />
      </Form.Item>
      <Button type="primary" onClick={handleSubmit}>
        Submit
      </Button>
      <Button type="primary" onClick={handleReset}>
        Reset
      </Button>
      <ProTable
        columns={columns}
        dataSource={data}
        pagination={{
          showQuickJumper: true,
        }}
        search={false}
        form={form}
      />
    </Form>
  );
};

export default App;
```
