如果您使用的是 Ant Design 的表格组件，可以通过设置表头的 `title` 属性为表头内容来实现在tooltip中显示表头文本。示例代码如下：

```jsx
import { Table } from "antd";

const columns = [
  {
    title: "表头文本",
    dataIndex: "dataIndexName",
    ellipsis: true,
    // 添加title属性
    title:
      "这是一个非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常长的表头文本",
  },
  {
    title: "数据",
    dataIndex: "data",
  },
];

const data = [
  {
    key: 1,
    dataIndexName:
      "这是一个非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常非常长的表头文本",
    data: "数据1",
  },
  {
    key: 2,
    dataIndexName: "表头文本2",
    data: "数据2",
  },
];

function App() {
  return <Table columns={columns} dataSource={data} />;
}
```

更多关于 Ant Design 表格组件的文档请参考 [Ant Design Table](https://ant.design/components/table-cn/)。
