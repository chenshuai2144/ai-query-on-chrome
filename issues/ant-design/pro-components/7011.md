## SuperZhanggy

ProTable 组件官方文档的demo 里有，类似以下的设置

// 设置某些列隐藏
const [columnsStateMap, setColumnsStateMap] = useState<Record<string, ColumnsState>>({
option: {
show: false,
order: 2,
},
});

<ProTable
columnsState={{
                              value: columnsStateMap,
                              onChange: setColumnsStateMap,
                          }}
/>
