可以通过监听 ProTable 的 `onPageIndexChange` 事件来实现，当页码发生变化时，再进行删除操作。具体实现如下：

```jsx
import { ProTable, useTable } from 'antd';

const columns = [
  {
    title: 'Name',
    dataIndex: 'name',
    key: 'name',
  },
  {
    title: 'Age',
    dataIndex: 'age',
    key: 'age',
  },
];

const data = [
  {
    key: '1',
    name: 'John Brown',
    age: 32,
  },
  {
    key: '2',
    name: 'Jim Green',
    age: 42,
  },
  {
    key: '3',
    name: 'Joe Black',
    age: 32,
  },
];

const App: React.FC = () => {
  const { tableRef, pagination, loading, getTableData } = useTable({
    columns,
    data,
    pagination: {
      pageSize: 5,
      showTotal: (total, page) => `Total: ${total} (${page} pages)`,
    },
  });

  const handleDelete = (record: any) => {
    // 删除操作
    console.log(record);
    // 更新数据
    getTableData();
  };

  const handlePageIndexChange = (page: number) => {
    // 当页码发生变化时，进行删除操作
    if (page === 1) {
      handleDelete(tableRef.current.data[0]);
    }
  };

  return (
    <ProTable
      ref={tableRef}
      rowKey="key"
      pagination={pagination}
      loading={loading}
      dataSource={getTableData()}
      columns={columns}
      onPageIndexChange={handlePageIndexChange}
    />
  );
};

export default App;
```

在这个例子中，当页码变为 1 时，会触发 `handlePageIndexChange` 方法，然后在该方法中调用 `handleDelete` 方法进行删除操作。
