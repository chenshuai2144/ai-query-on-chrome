## cnyballk

```ts
// .umirc.ts

export default {
  chainWebpack(config) {
    const GLSL_REG = /\.(glsl|vs|fs)$/;

    config.module.rule("asset").exclude.add(GLSL_REG).end();

    config.module
      .rule("glslify")
      .test(GLSL_REG)
      .exclude.add(/node_modules/)
      .end()
      .use("raw-loader")
      .loader("raw-loader")
      .end()
      .use("glslify-loader")
      .loader("glslify-loader")
      .end();
  },
};
```
